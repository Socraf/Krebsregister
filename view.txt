* <SIGNATURE>---------------------------------------------------------------------------------------+
* | Instance Private Method /KAP/KR_CL_VIEW->GET_CSS_BOOTSTRAP
* +-------------------------------------------------------------------------------------------------+
* +--------------------------------------------------------------------------------------</SIGNATURE>
  METHOD get_css_bootstrap.

    TRY.

        IF gr_html_ctrl IS NOT INITIAL.

          "Bootstrap css laden
          gr_html_ctrl->load_mime_object(
            EXPORTING
              object_id  = '/KAP/BASE_BS46CSS'
              object_url = 'BOOTSTRAP.CSS'
          ).
          IF sy-subrc <> 0.
            log_message(
              EXPORTING
                i_msg_text    = 'Objekt Bootstrap CSS nicht vorhanden.'
                i_msg_type    = 'W'
                i_method_name = 'PRESET_CSS'
                ir_item       = me
            ) ##NO_TEXT.
          ELSE.

            gr_web_doc->head->create_link_child( )->set_href( i_value = 'BOOTSTRAP.CSS'  )->set_rel( i_value = 'stylesheet'  ).

          ENDIF.

        ENDIF.

      CATCH cx_dynamic_check INTO exception.

        log_exception( ).

    ENDTRY.

  ENDMETHOD.

* <SIGNATURE>---------------------------------------------------------------------------------------+
* | Instance Private Method /KAP/KR_CL_VIEW->GET_HTML_BODY_DOCUMENT
* +-------------------------------------------------------------------------------------------------+
* | [--->] I_SAPEVENT                     TYPE        STRING
* | [--->] I_IDFORM                       TYPE        STRING
* | [--->] I_IDBTN                        TYPE        STRING
* | [--->] I_TXTBTN                       TYPE        STRING
* | [--->] IS_KRDDOCS                     TYPE        /KAP/KR_ST_KRDDOCS
* | [--->] I_MODE                         TYPE        NUMC1
* | [<-()] R_HTML                         TYPE        STRING
* +--------------------------------------------------------------------------------------</SIGNATURE>
  METHOD get_html_body_document.

* I_MODE: Hilfsvariable;
*   1 KRO => Operation
*   2 KRDST => Strahltherapie
*   3 KRSY => Systemische Therapie
*   4 KRV => Verlauf
*   5 KRTK => Tumorkonferenz

    DATA: l_krodocs       TYPE string,
          l_datum         TYPE string,
          ls_krddocs_head TYPE /kap/kr_st_krddocs,
          ls_krddocs_body TYPE /kap/kr_st_krddocs,
          lt_krddocs_body TYPE /kap/kr_tt_krddocs.

    CASE i_mode.

      WHEN 1. "KRO => Operation

        ls_krddocs_head = is_krddocs.
        ls_krddocs_head-dtid = /kap/kr_cl_model=>g_kro_dtid.
        ls_krddocs_head-dtvers = /kap/kr_cl_model=>get_latest_version_document(  ls_krddocs_head-dtid ).
        lt_krddocs_body = gt_krodocs.

      WHEN 2. "KRDST => Strahltherapie

        ls_krddocs_head = is_krddocs.
        ls_krddocs_head-dtid = /kap/kr_cl_model=>g_krst_dtid.
        ls_krddocs_head-dtvers = /kap/kr_cl_model=>get_latest_version_document(  ls_krddocs_head-dtid ).
        lt_krddocs_body = gt_krstdocs.

      WHEN 3. "KRSY => Systemische Therapie

        ls_krddocs_head = is_krddocs.
        ls_krddocs_head-dtid = /kap/kr_cl_model=>g_krsy_dtid.
        ls_krddocs_head-dtvers = /kap/kr_cl_model=>get_latest_version_document(  ls_krddocs_head-dtid ).
        lt_krddocs_body = gt_krsydocs.

      WHEN 4. "KRV => Verlauf

        ls_krddocs_head = is_krddocs.
        ls_krddocs_head-dtid = /kap/kr_cl_model=>g_krv_dtid.
        ls_krddocs_head-dtvers = /kap/kr_cl_model=>get_latest_version_document(  ls_krddocs_head-dtid ).
        lt_krddocs_body = gt_krvdocs.


      WHEN 5. "KRTK => Tumorkonferenz

        ls_krddocs_head = is_krddocs.
        ls_krddocs_head-dtid = /kap/kr_cl_model=>g_krtk_dtid.
        ls_krddocs_head-dtvers = /kap/kr_cl_model=>get_latest_version_document(  ls_krddocs_head-dtid ).
        lt_krddocs_body = gt_krtkdocs.

      WHEN OTHERS.

    ENDCASE.

    r_html =
           "Neue Dokument erzeugen
           '<div class="col">'
        && '<form action="SAPEVENT:' && i_sapevent && '" accept-charset="utf-8" method="post" name="' && i_idform && '" id="' && i_idform && '">'
        && me->get_html_input_hidden( i_mode = 2 is_krddocs = ls_krddocs_head ) "Dokumentendaten teilen
        &&  '<button class="btn btn-outline-primary btn-block btn-head" id="' && i_idbtn && '" name="' && i_idbtn && '" type="submit">'
        &&   i_txtbtn  "&& ' &#10133;'
        &&  '</button>'
        && '</form>'.


    "Vorhandene Dokumente auflisten
    LOOP AT lt_krddocs_body INTO ls_krddocs_body.

      IF ls_krddocs_body-dkey1 EQ is_krddocs-dkey1.

        l_datum = /kap/kr_cl_model=>get_message( '049' ).

        CONCATENATE  l_datum ls_krddocs_body-dodat_txt INTO l_datum SEPARATED BY space. "Meldung von 01.01.1990 z.B

        l_krodocs = l_krodocs
          && '<form action="SAPEVENT:ON_OPEN_KRD" accept-charset="utf-8" method="post" name="OPEN_KRO" id="OPEN_KRO">'
          && me->get_html_input_hidden( i_mode = 3 is_krddocs = ls_krddocs_body ) "Dokumentenschlüssel teilen
          && '<ul class="list-group list-group-flush text-left">'
          &&   '<button class="btn btn-link btn-list" id="new_icd" name="OPEN_KRO" type="submit">'
          &&   l_datum
          &&   '</button>'
          && '</ul>'
          && '</form>'.

      ENDIF.

      CLEAR: ls_krddocs_body, l_datum.

    ENDLOOP.

    r_html = r_html
    && l_krodocs
    && '</div>'.

    CLEAR:  l_krodocs, lt_krddocs_body.

  ENDMETHOD.

* <SIGNATURE>---------------------------------------------------------------------------------------+
* | Instance Public Method /KAP/KR_CL_VIEW->SET_START_HML_VALUES
* +-------------------------------------------------------------------------------------------------+
* | [--->] IT_QUERY                       TYPE        CNHT_QUERY_TABLE
* +--------------------------------------------------------------------------------------</SIGNATURE>
  METHOD set_start_hml_values.

    DATA: ls_query TYPE w3query.

    IF it_query IS INITIAL.

      RETURN.

    ENDIF.

    LOOP AT it_query INTO ls_query.

      CASE ls_query-name.

        WHEN 'einri_in'.

          gs_start_html-einri = ls_query-value.

        WHEN 'pat_nr_in'.

          gs_start_html-patient = ls_query-value.

        WHEN 'fall_nr_in'.

          gs_start_html-fall = ls_query-value.

        WHEN 'pat_nname_in'.

          gs_start_html-nname = ls_query-value.

        WHEN 'pat_vname_in'.

          gs_start_html-vname = ls_query-value.

        WHEN 'pat_gname_in'.

          gs_start_html-gname = ls_query-value.

        WHEN 'pat_gesch_in'.

          gs_start_html-geschl = ls_query-value.

        WHEN 'pat_gebdat_in'.

          gs_start_html-gebdat = ls_query-value.

      ENDCASE.

    ENDLOOP.

    g_preallocate_organisation = 'X'.

  ENDMETHOD.

* <SIGNATURE>---------------------------------------------------------------------------------------+
* | Instance Public Method /KAP/KR_CL_VIEW->UPDATE_ICD_CANCER_REGISTRY
* +-------------------------------------------------------------------------------------------------+
* | [--->] IT_QUERY_TABLE                 TYPE        CNHT_QUERY_TABLE
* +--------------------------------------------------------------------------------------</SIGNATURE>
  METHOD UPDATE_ICD_CANCER_REGISTRY.

    DATA: ls_query   TYPE w3query,
          ls_icdlist TYPE /kap/kr_st_icdlist,
          l_dkey     TYPE diaid.

    READ TABLE it_query_table INTO ls_query WITH KEY name = 'icd_input'.

    IF sy-subrc EQ 0 AND ls_query-value IS NOT INITIAL.

      l_dkey = ls_query-value.

      READ TABLE gt_icdlist INTO ls_icdlist WITH KEY dkey1 = l_dkey.

      IF sy-subrc EQ  0.

        gs_krebsregister-dkey1  = ls_icdlist-dkey1.
        gs_krebsregister-dkat1  = ls_icdlist-dkat1.
        gs_krebsregister-dtext1 = ls_icdlist-ditxt.
        gs_krebsregister-dialo  = ls_icdlist-dialo.
        gs_krebsregister-diadt  = ls_icdlist-diadt.

      ENDIF.

    ENDIF.

  ENDMETHOD.
